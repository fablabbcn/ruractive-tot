## Possible solution!

### Arduino code

```c++
#include <Temperature_LM75_Derived.h>

TI_TMP102 temperature;

const int push = 0;
bool pressed = false;

void setup() {
  pinMode(push, INPUT);
  Serial.begin(115200);
  Wire.begin();
}

void loop() {
  if (!digitalRead(push) && !pressed) {
    pressed = true;
    Serial.print("Temperature = ");
    Serial.print(temperature.readTemperatureC());
    Serial.println(" C");
  } else if (digitalRead(push)) {
    pressed = false;
  }
}
```
